1 /**
  2  * @namespace A class to contain tool definitions and associated tool methods for user interaction.
  3  */
  4 var Tool = {			
  5 	initialize: function() {
  6 		// default tool on startup is the Pan tool:
  7 		Glop.observe('mousemove', Tool.Pan.mousemove)
  8 		Glop.observe('mousedown', Tool.Pan.mousedown)
  9 		Glop.observe('mouseup', Tool.Pan.mouseup)
 10 	},
 11 	/**
 12 	 * The tool currently in use. Options include 'pan', 'pen', 'select'
 13 	 */
 14 	active: 'Pan',
 15 	/**
 16 	 * Function to change the active tool 
 17 	 */
 18 	change: function(new_tool) {
 19 		old_tool = Tool.active
 20 			
 21 		tool_events = ['mousemove','mouseup','mousedown']
 22 
 23 		tool_events.each(function(tool_event) {
 24 			Glop.stopObserving(tool_event,Tool[old_tool][tool_event])
 25 			Glop.observe(tool_event,Tool[new_tool][tool_event])
 26 		})
 27 
 28 		if (!Object.isUndefined(Tool[old_tool].deactivate)) {
 29 			Tool[old_tool].deactivate()
 30 		}
 31 		if (!Object.isUndefined(Tool[new_tool].activate)) {
 32 			Tool[new_tool].activate()
 33 		}
 34 		
 35 		Tool.active = new_tool
 36 	},
 37 	/*
 38 	 * Pass drag call to the active tool:
 39 	 */
 40 	drag: function() {
 41 		Tool[Tool.active].drag()
 42 	}
 43 }
