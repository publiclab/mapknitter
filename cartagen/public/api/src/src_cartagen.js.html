 44 /**
 45  * Array of all objects that should be drawn.
 46  * @type Feature[]
 47  * @deprecated
 48  */
 49 var objects = []

 51 // temp object unitl PhoneGap is initialized.
 52 PhoneGap = window.DeviceInfo && DeviceInfo.uuid != undefined
 53 
 54 if (typeof cartagen_base_uri == 'undefined') {
 55 	/**
 56 	 * Path to the cartagen directory. Defaults to "cartagen", which works only
 57 	 * if the cartagen directory is named "cartagen" and is located in the
 58 	 * same directoy as the current page. This should be set before Cartagen
 59 	 * is loaded if it needs to be changed.
 60 	 * 
 61 	 * @type String
 62 	 */
 63     cartagen_base_uri = 'cartagen'
 64 }
 65 
 66 // if (Prototype.Browser.MobileSafari) $('brief').hide()
 67 
 68 /**
 69  * @namespace
 70  * Namespace for methods and variables that manage Cartagen as a whole, i.e. loading
 71  * data and creating Nodes and Ways.
 72  */
 73 var Cartagen = {
 74 	/**
 75 	 * Queue of labels to draw
 76 	 * @type Label[]
 77 	 */
 78 	label_queue: [],
 79 	/**
 80 	 * Queue of features to be drawn at the end of draw()
 81 	 */
 82 	feature_queue: [],
 83 	/**
 84 	 * An array of scripts that will be loaded when Cartagen is initialized.
 85 	 * 
 86 	 * @type String[]
 87 	 * @see Cartagen.initialize
 88 	 */
 89 	scripts: [],
 90 	/**
 91 	 * Registers initialize to run with the given configs when window is loaded
 92 	 * @param {Object} configs A set of key/value pairs that will be copied to the Cartagen object
 93 	 */
 94 	setup: function(configs) {
 95 		$(document).observe('dom:loaded', function() {
 96 			$('canvas').insert('<canvas id="main"></canvas>')
 97 			Cartagen.initialize(configs)
 98 		})	
 99 	},
100 	/**
101 	 * Performs initialization tasks, mainly fetching map data. This should never be called directly,
102 	 * rather it is intended to be called by setup
103 	 * @param {Object} configs A set of key/value pairs that will be copied to the Cartagen object
104 	 */
105 	initialize: function(configs) {
106 		Config.init(configs)
107 		// add a new layer for dynamic drawing (hovers, clicks)
108 		// $C.add('background')
109 		
110 		// load phonegap js if needed
111 		if (window.PhoneGap) {
112 			Cartagen.scripts.unshift(cartagen_base_uri + '/lib/phonegap/phonegap.base.js',
113 						             cartagen_base_uri + '/lib/phonegap/geolocation.js',
114 						             cartagen_base_uri + '/lib/phonegap/iphone/phonegap.js',
115 						             cartagen_base_uri + '/lib/phonegap/iphone/geolocation.js')
116 		}
117 
118 		// load extra scripts:
119 		Cartagen.load_next_script()
120 		
121 		// browser stuff:
122 		this.browser_check()
123 		
124 		//if (Prototype.Browser.MobileSafari) window.scrollTo(0, 1) //get rid of url bar
125 
126 		/**
127 		 * @name Cartagen#cartagen:init
128 		 * @event
129 		 * Fired after Cartagen loads its configuration and the contents of 
130 		 * Cartagen.scripts, but before any features are loaded or drawn.
131 		 * Note that Cartagen.scripts are loaded asynchronously, so it is not
132 		 * guarenteed that they will be finished loading.
133 		 */
134 		document.fire('cartagen:init')
135 		
136 		// bind event listeners
137 		Glop.observe('glop:draw', Cartagen.draw.bindAsEventListener(this))
138 		Glop.observe('glop:postdraw', Cartagen.post_draw.bindAsEventListener(this))
139 
140 		// Startup:
141 		Style.load_styles(Config.stylesheet) // stylesheet
142 		
143 		if (!Config.static_map) {
144 			Importer.get_current_plot(true)
145 			new PeriodicalExecuter(Glop.trigger_draw,3)
146 			new PeriodicalExecuter(function() { Importer.get_current_plot(false) },3)
147 		} else {
148 			Config.static_map_layers.each(function(layer_url) {
149 				$l('fetching '+layer_url)
150 				Importer.get_static_plot(layer_url)
151 			},this)
152 			// to add user-added map data... messy!
153 			if (Config.dynamic_layers.length > 0) {
154 				Config.dynamic_layers.each(function(layer_url) {
155 					$l('fetching '+layer_url)
156 					load_script(layer_url)
157 				},this)
158 			}
159 		}
160 		
161 		Glop.trigger_draw()
162 		
163 		/**
164 		 * @name cartagen:postinit
165 		 * @event
166 		 * Fired after Caragen loads map data.
167 		 */
168 		document.fire('cartagen:postinit')
169 	},
170 	/**
171 	 * Runs every frame in the draw() method. An attempt to isolate cartagen code from general GLOP code.
172 	 * Uses Geohash to draw each feature on the map.
173 	 * @param {Event} e The Glop draw event.
174 	 */
175 	draw: function(e) {
176 		e.no_draw = true
177 
178 		// if (Prototype.Browser.MobileSafari || window.PhoneGap) Config.simplify = 2
179 		Style.style_body()
180         
181 		// display fps if Config.fps = true
182 		if (Config.fps) {
183 			if ($('cartagen_fps')) {
184 				$('cartagen_fps').innerHTML = Glop.fps
185 			} else {
186 				$$('body')[0].insert({top:'<div style="position:absolute;margin:10px;font-weight:bold;background:white" id="cartagen_fps"></div>'})
187 			}
188 		}
189 
190 		// $C.canvases.keys().each(function(canvas) {
191 		// 	$C.open(canvas)
192 			$C.translate(Glop.width / 2, Glop.height / 2)
193 			        $C.rotate(Map.rotate)
194 			        $C.scale(Map.zoom, Map.zoom)
195 			        $C.translate(-Map.x,-Map.y)
196 		// })
197 		// 
198 		// $C.close()
199         
200 		Viewport.draw() //adjust viewport
201 		
202 		/**
203 		 *@name Cartagen#cartagen:predraw
204 		 *Fires just before features are drawn
205 		 *@event
206 		 */
207 		Glop.fire('cartagen:predraw')
208 		
209 		//Geohash lookup:
210 		Geohash.objects.each(function(object) {
211 			if (object.user_submitted) {
212 				Cartagen.feature_queue.push(object)
213 			}
214 			else {
215 				// try {
216 				object.draw()
217 				// } catch(e) {$l(e)}
218 			}
219 		})
220 
221 		this.feature_queue.each(function(item) {
222 			(item.draw.bind(item))()
223 		})
224 		this.feature_queue = []
225 		
226 		if (Prototype.Browser.MobileSafari || window.PhoneGap) User.mark()
227 	},
228     /**
229      * Runs every frame, after everything else has been done.
230      */
231     post_draw: function() {
232         this.label_queue.each(function(item) {
233             item[0].draw(item[1], item[2])
234         })
235 
236 		this.label_queue = []
237 
238 		/**
239 		 *@name Cartagen#cartagen:predraw
240 		 *Fires just after labels are drawn
241 		 *@event
242 		 */
243 		Glop.fire('cartagen:postdraw')
244 		
245 		// display percentage of features we've imported so far:
246 		// $C.close()
247 		Interface.display_loading(Importer.parse_manager.completed)
248 		
249     },
250     /**
251      * Adds the label to the list of labels to be drawn during post_draw
252      * @param {Label}  label The label to draw
253      * @param {Number} x     x-coord
254      * @param {Number} y     y-coord
255      */
256     queue_label: function(label, x, y) {
257         this.label_queue.push([label, x, y])
258     },
259 	/**
260 	 * Show alert if it's IE.
261 	 */
262 	browser_check: function() {
263 		if ($('browsers')) {
264 			$('browsers').absolutize();
265 			$('browsers').style.top = "100px";
266 			$('browsers').style.margin = "0 auto";
267 			if (Prototype.Browser.IE) $('browsers').show();
268 		}
269 	},
270 	/**
271 	 * Repositions Cartagen to center on lat,lon with given zoom level
272      * @param {Number} lat     latitude to center the map on
273      * @param {Number} lon     longitude to center the map on
274      * @param {Number} zoom_level     zoom_level to set the map to
275 	 */
276 	go_to: function(lat,lon,zoom_level) {
277 		Map.zoom = zoom_level
278 		Map.lat = lat
279 		Map.lon = lon
280 		Map.x = Projection.lon_to_x(Map.lon)
281 		Map.y = Projection.lat_to_y(Map.lat)
282 	},
283 	/**
284 	 * Searches all objects by tags, and sets highlight=true for all matches.
285 	 * 
286 	 * @param {Object} query The tag to search for
287 	 */
288 	highlight: function(query) {
289 		Geohash.objects.each(function(object) {
290 			object.highlight = false
291 			if (query != "" && object.tags && object instanceof Way) {
292 				object.tags.each(function(tag) {
293 					if (tag.key.toLowerCase().match(query.toLowerCase()) || tag.value.toLowerCase().match(query.toLowerCase())) {
294 						object.highlight = true
295 					}
296 				})
297 				if (object.user && object.user.toLowerCase().match(query.toLowerCase())) object.highlight = true
298 				if (object.description && object.description.toLowerCase().match(query.toLowerCase())) object.highlight = true
299 			}
300 		})
301 	},
302 	/**
303 	 * Shows the live GSS editor. Generally only for cartgen.org.
304 	 */
305 	show_gss_editor: function() {
306 		$('description').hide()
307 		$('brief').style.width = '28%'
308 		$('brief_first').style.width = '92%';
309 		$('gss').toggle()
310 		Config.live_gss = !Config.live_gss
311 	},
312 	/**
313 	 * Sends user to an image of the current canvas
314 	 */
315 	redirect_to_image: function() {
316 		try {
317 				window.open($C.to_data_url())
318 			} catch(e) {
319 				alert("Sorry, this stylesheet uses remote images; JavaScript does not allow these to be used to generate an image.")
320 		}
321 	},
322 	/**
323 	 * Loads each script in scripts array, sequentially.
324 	 * Requires a load_next_script() call at the end of each
325 	 * dependent script to trigger the next one.
326 	 */
327 	load_next_script: function() {
328 		$l("loading: "+Cartagen.scripts[0])
329 		if (Cartagen.scripts.length > 0) {
330 			Cartagen.load_script(Cartagen.scripts.splice(0,1)[0])
331 		}
332 	},
333 	/**
334 	 * Loads a script into <script src="" /> tags, no cross-domain limits.
335 	 * @param {String} script Path to the script
336 	 */
337 	load_script: function(script) {
338 		$$('head')[0].insert(new Element('script', { 
339 			'src': script, 
340 			'type': 'text/javascript', 
341 			'charset': 'utf-8', 
342 			evalJSON: 'force' 
343 		}));
344 	},
345 	/**
346 	 * Loads data from a local KML file and dumps it into the Geohash object.
347 	 * @param {String} url Path to the data file
348 	 */
349 	import_kml: function(url) {
350 		new Ajax.Request(url,{
351 			method: 'get',
352 			onComplete: function(result) {
353 				$l('completed load of KML')
354 				response = result
355 				$l(xml2json.xml_to_object(result.responseText))
356 				$l('completed import of KML')
357 			}
358 		})
359 		
360 	}
361 }
